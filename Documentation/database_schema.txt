-- ─────────────────────────────────────────
-- [TABLE 1] profiles
-- 1:1 with auth.users. Stores display info, avatar, demographics.
-- ─────────────────────────────────────────
create table public.profiles (
    id          uuid        not null primary key references auth.users (id) on delete cascade,
    full_name   text,
    avatar_url  text,
    dob         date,
    gender      text,
    country     text,
    created_at  timestamptz not null default timezone('utc', now()),
    updated_at  timestamptz
);

-- ─────────────────────────────────────────
-- [TABLE 2] sessions
-- Top-level record for each live wingman or consultant session.
-- mode:   'live_wingman' | 'consultant'
-- status: 'active' | 'completed'
-- ─────────────────────────────────────────
create table public.sessions (
    id          uuid        not null default extensions.uuid_generate_v4() primary key,
    user_id     uuid        not null references auth.users (id) on delete cascade,
    mode        text        not null default 'consultant'
                            check (mode in ('live_wingman', 'consultant')),
    status      text        not null default 'active'
                            check (status in ('active', 'completed')),
    title       text                 default 'New Conversation',
    summary     text,
    created_at  timestamptz not null default timezone('utc', now()),
    ended_at    timestamptz
);

-- ─────────────────────────────────────────
-- [TABLE 3] session_logs
-- Full conversation transcript. role: 'user' | 'others' | 'llm' | 'system'
-- Realtime-enabled so Flutter receives new rows instantly during live sessions.
-- ─────────────────────────────────────────
create table public.session_logs (
    id              uuid        not null default extensions.uuid_generate_v4() primary key,
    session_id      uuid        not null references public.sessions (id) on delete cascade,
    role            text        not null check (role in ('user', 'others', 'llm', 'system')),
    content         text        not null,
    speaker_label   text,
    confidence      float,
    created_at      timestamptz not null default timezone('utc', now())
);

-- ─────────────────────────────────────────
-- [TABLE 4] consultant_logs
-- Lightweight Q&A log for consultant mode history (drawer).
-- session_id groups messages into a single chat thread.
-- ─────────────────────────────────────────
create table public.consultant_logs (
    id          uuid        not null default extensions.uuid_generate_v4() primary key,
    user_id     uuid        not null references auth.users (id) on delete cascade,
    session_id  uuid                 references public.sessions (id) on delete set null,
    question    text,
    answer      text,
    created_at  timestamptz not null default timezone('utc', now())
);

-- ─────────────────────────────────────────
-- [TABLE 5] memory
-- Long-term vector memory. embedding: 384-dim MiniLM-L6-v2.
-- memory_type: 'fact' | 'summary' | 'event' | 'preference'
-- ─────────────────────────────────────────
create table public.memory (
    id          uuid        not null default extensions.uuid_generate_v4() primary key,
    user_id     uuid        not null references auth.users (id) on delete cascade,
    session_id  uuid                 references public.sessions (id) on delete set null,
    content     text        not null,
    memory_type text                 default 'fact'
                            check (memory_type in ('fact', 'summary', 'event', 'preference')),
    embedding   extensions.vector(384),
    created_at  timestamptz not null default timezone('utc', now())
);

-- ─────────────────────────────────────────
-- [TABLE 6] knowledge_graphs
-- Serialised NetworkX graph as JSONB. One record per user.
-- node_count: generated from the nodes array (NULL = no graph yet).
-- ─────────────────────────────────────────
create table public.knowledge_graphs (
    user_id     uuid        not null primary key references auth.users (id) on delete cascade,
    graph_data  jsonb,
    node_count  int         generated always as (jsonb_array_length(graph_data->'nodes')) stored,
    updated_at  timestamptz          default now()
);

-- ─────────────────────────────────────────
-- [TABLE 7] entities
-- Named entity registry. One row per unique entity per user.
-- canonical_name: lowercase normalised deduplication key.
-- entity_type: 'person' | 'place' | 'organization' | 'event' | 'object' | 'concept'
-- ─────────────────────────────────────────
create table public.entities (
    id              uuid        not null default extensions.uuid_generate_v4() primary key,
    user_id         uuid        not null references auth.users (id) on delete cascade,
    canonical_name  text        not null,
    display_name    text        not null,
    entity_type     text        not null default 'person'
                                check (entity_type in ('person','place','organization','event','object','concept')),
    description     text,
    first_seen_at   timestamptz not null default timezone('utc', now()),
    last_seen_at    timestamptz not null default timezone('utc', now()),
    mention_count   int         not null default 1,
    created_at      timestamptz not null default timezone('utc', now()),
    constraint entities_user_canonical_unique unique (user_id, canonical_name)
);

-- ─────────────────────────────────────────
-- [TABLE 8] entity_attributes
-- Key-value facts per entity. One value per key (unique).
-- ─────────────────────────────────────────
create table public.entity_attributes (
    id              uuid        not null default extensions.uuid_generate_v4() primary key,
    entity_id       uuid        not null references public.entities (id) on delete cascade,
    attribute_key   text        not null,
    attribute_value text        not null,
    source_session  uuid                 references public.sessions (id) on delete set null,
    confidence      float                default 1.0,
    created_at      timestamptz not null default timezone('utc', now()),
    updated_at      timestamptz not null default timezone('utc', now()),
    constraint entity_attr_unique unique (entity_id, attribute_key)
);

-- ─────────────────────────────────────────
-- [TABLE 9] entity_relations
-- Directed edges: (source) --[relation]--> (target)
-- ─────────────────────────────────────────
create table public.entity_relations (
    id              uuid        not null default extensions.uuid_generate_v4() primary key,
    user_id         uuid        not null references auth.users (id) on delete cascade,
    source_id       uuid        not null references public.entities (id) on delete cascade,
    target_id       uuid        not null references public.entities (id) on delete cascade,
    relation        text        not null,
    weight          float                default 1.0,
    source_session  uuid                 references public.sessions (id) on delete set null,
    is_verified     bool                 default false,
    created_at      timestamptz not null default timezone('utc', now()),
    updated_at      timestamptz not null default timezone('utc', now()),
    constraint entity_relations_unique unique (source_id, target_id, relation)
);

-- ─────────────────────────────────────────
-- [TABLE 10] highlights
-- Home-screen flags: conflicts, important facts, suggestions, warnings.
-- highlight_type: 'conflict' | 'important' | 'suggestion' | 'warning'
-- Realtime-enabled.
-- ─────────────────────────────────────────
create table public.highlights (
    id              uuid        not null default extensions.uuid_generate_v4() primary key,
    user_id         uuid        not null references auth.users (id) on delete cascade,
    session_id      uuid                 references public.sessions (id) on delete set null,
    highlight_type  text        not null default 'important'
                                check (highlight_type in ('conflict','important','suggestion','warning')),
    title           text        not null,
    body            text        not null,
    related_entity  uuid                 references public.entities (id) on delete set null,
    is_read         bool        not null default false,
    is_resolved     bool        not null default false,
    is_dismissed    bool        not null default false,
    created_at      timestamptz not null default timezone('utc', now())
);

-- ─────────────────────────────────────────
-- [TABLE 11] events
-- Calendar items and deadlines extracted from conversations.
-- Realtime-enabled.
-- ─────────────────────────────────────────
create table public.events (
    id              uuid        not null default extensions.uuid_generate_v4() primary key,
    user_id         uuid        not null references auth.users (id) on delete cascade,
    session_id      uuid                 references public.sessions (id) on delete set null,
    title           text        not null,
    description     text,
    due_at          timestamptz,
    due_text        text,
    related_entity  uuid                 references public.entities (id) on delete set null,
    is_completed    bool        not null default false,
    created_at      timestamptz not null default timezone('utc', now())
);

-- ─────────────────────────────────────────
-- [TABLE 12] audio_sessions
-- Device-side audio recording metadata. Audio binary never leaves the device.
-- ─────────────────────────────────────────
create table public.audio_sessions (
    id                uuid        not null default extensions.uuid_generate_v4() primary key,
    session_id        uuid        not null references public.sessions (id) on delete cascade,
    user_id           uuid        not null references auth.users (id) on delete cascade,
    local_path        text        not null,
    file_name         text        not null,
    format            text                 default 'aac'
                                 check (format in ('aac', 'm4a', 'wav', 'mp3')),
    duration_seconds  float,
    file_size_bytes   bigint,
    device_id         text,
    recorded_at       timestamptz not null default timezone('utc', now()),
    is_available      bool        not null default true

-- [TABLE 13] voice_enrollments
-- Speaker fingerprint. One row per user.

create table public.voice_enrollments (
    id              uuid        not null default extensions.uuid_generate_v4() primary key,

    model_version   text                 default 'v1',

-- Vector similarity search over long-term memory.
-- Called by the server before every LLM prompt to inject relevant past facts.
create or replace function public.match_memory(
    query_embedding  extensions.vector(384),
    match_threshold  float,
    match_count      int,
    p_user_id        uuid
)
returns table (id uuid, content text, similarity float)
language sql stable
as $$
    select  id,
            content,
            1 - (embedding <=> query_embedding) as similarity
    from    public.memory
    where   user_id = p_user_id
      and   1 - (embedding <=> query_embedding) > match_threshold
    order   by embedding <=> query_embedding
    limit   match_count;
$$;

knowledge_graphs.node_count
  Generated column calls jsonb_array_length(graph_data->'nodes').

  Called on the server with the Supabase auth UUID (not the LiveKit identity).
